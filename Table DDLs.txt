-- Trigger: recad_persona_tr_uniquecpf

-- DROP TRIGGER IF EXISTS recad_persona_tr_uniquecpf ON recadastra.recad_persona;

CREATE TRIGGER recad_persona_tr_uniquecpf
    BEFORE INSERT OR UPDATE OF cpf
    ON recadastra.recad_persona
    FOR EACH ROW
    EXECUTE FUNCTION public.fntg_recad_persona_tr_uniquecpf();



-- public.fntg_criar_anexos

BEGIN
  insert into recadastra.recad_anexo(fkpersona,fkanexotipo,obrigatorio,emissao_data,validade_data,cadastro_fkusuario,cadastro_data,url,arquivo,observacao)
  select 
        ps.id fkpersona,
        at.id fkanexotipo,
        at.obrigatorio obrigatorio,
        null emissao_data,
        null validade_data,
        null cadastro_fkusuario,
        null cadastro_data,
        null url,
        null arquivo,
        null observacao
    from recadastra.recad_persona ps
    join recadastra.recad_anexotipo at on true
   where not exists (select 1 from recadastra.recad_anexo ax where ax.fkpersona = ps.id and ax.fkanexotipo = at.id) 
     and at.obrigatorio = 1;
   return new;
END;
